#include <stdio.h>
#include <stdlib.h>

int main(void)
{
  
  int c;
  for(;;){  // wykonywanie nieskończenie wiele razy

    printf("\nWprowadź cyfrę od 0 do 5: \n");  
    scanf(" %d",&c); // wielokrotne wczytywanie pojedynczego znaku (jak pozbyć się \n ?)
    if(c==0){
    printf("Wprowadzono: 0\n");}
    else if(c==1){
    printf("Wprowadzono: 1\n");}
    else if(c==2){
    printf("Wprowadzono: 2\n");}
    else if(c==3){
    printf("Wprowadzono: 3\n");}
    else if(c==4){
    printf("Wprowadzono: 4\n");}
    else if(c==5){
    printf("Wprowadzono: 5\n");}
    else if (c==6){
    exit(0);}
    else {
    printf("Wprowadzono: znak spoza zakresu 0-6\n");}
    
    
  
    
  }

}
